function doGet(e) { const ss = SpreadsheetApp.getActiveSpreadsheet(); const action = e.parameter.action || ''; if (action === 'needs-wall') { const sheet = ss.getSheetByName('needswall'); if (!sheet) return ContentService.createTextOutput(JSON.stringify({ entries: [] })).setMimeType(ContentService.MimeType.JSON); const rows = sheet.getDataRange().getValues(); const entries = rows.slice(1).map(r => ({ firstName: r[0], description: r[1], situation: r[2], prayer: r[3], status: r[4] })).filter(e => e.status === 'approved' || e.status === 'met'); return ContentService.createTextOutput(JSON.stringify({ entries })).setMimeType(ContentService.MimeType.JSON); } if (action === 'needs-wall-all') { const sheet = ss.getSheetByName('needswall'); if (!sheet) return ContentService.createTextOutput(JSON.stringify({ entries: [] })).setMimeType(ContentService.MimeType.JSON); const rows = sheet.getDataRange().getValues(); const entries = rows.slice(1).map(r => ({ firstName: r[0], description: r[1], situation: r[2], prayer: r[3], status: r[4], submitter: r[5] || '', submitterPhone: r[6] || '', submitterEmail: r[7] || '', lastName: r[8] || '', theirPhone: r[9] || '', theirEmail: r[10] || '' })); return ContentService.createTextOutput(JSON.stringify({ entries })).setMimeType(ContentService.MimeType.JSON); } return ContentService.createTextOutput(JSON.stringify({ ok: true })).setMimeType(ContentService.MimeType.JSON); } function doPost(e) { try { const ss = SpreadsheetApp.getActiveSpreadsheet(); const data = JSON.parse(e.postData.contents); const NOTIFY_EMAIL = 'hello@thejesusbarn.org'; Logger.log('doPost action: ' + (data.action || data.sheet || 'unknown')); if (row[0] === 'raised') sheet.getRange(i+1, 2).setValue(data.raised); if (row[0] === 'contributors') sheet.getRange(i+1, 2).setValue(data.contributors); }); } // ── NEEDS ── if (data.sheet === 'needs') { const sheet = ss.getSheetByName('needs'); sheet.clearContents(); sheet.getRange(1, 1).setValue('item'); sheet.getRange(1, 2).setValue('status'); data.needs.forEach((need, i) => { sheet.getRange(i+2, 1).setValue(need.item); sheet.getRange(i+2, 2).setValue(need.status); }); } // ── WALL SUBMIT ── if (data.sheet === 'wall-submit') { const sheet = ss.getSheetByName('wall'); const lastRow = Math.max(sheet.getLastRow(), 1); sheet.getRange(lastRow+1, 1).setValue(data.name); sheet.getRange(lastRow+1, 2).setValue(data.location); sheet.getRange(lastRow+1, 3).setValue(data.message); sheet.getRange(lastRow+1, 4).setValue('pending'); MailApp.sendEmail({ to: NOTIFY_EMAIL, subject: '💬 New Community Wall Submission — ' + data.name, htmlBody: `

New Needs Wall Nomination

Pending your approval — The Jesus Barn

Submitted By
Name${data.submitter || '—'}

` }); } catch(mailErr) { Logger.log('Mail error: ' + mailErr.toString()); } return ContentService.createTextOutput(JSON.stringify({ ok: true })) .setMimeType(ContentService.MimeType.JSON); } // ── NEEDS WALL UPDATE (approve) ── if (data.action === 'needs-wall-update') { const sheet = ss.getSheetByName('needswall'); const rows = sheet.getDataRange().getValues(); for (let i = 1; i < rows.length; i++) { if (rows[i][0] === data.name || rows[i][1] === data.description) { sheet.getRange(i + 1, 5).setValue(data.status); break; } } return ContentService.createTextOutput(JSON.stringify({ ok: true })) .setMimeType(ContentService.MimeType.JSON); } // ── NEEDS WALL DELETE ── if (data.action === 'needs-wall-delete') { const sheet = ss.getSheetByName('needswall'); const rows = sheet.getDataRange().getValues(); for (let i = rows.length - 1; i >= 1; i--) { if (rows[i][0] === data.name || rows[i][1] === data.description) { sheet.deleteRow(i + 1); break; } } return ContentService.createTextOutput(JSON.stringify({ ok: true })) .setMimeType(ContentService.MimeType.JSON); } // ── SIGNUP ── if (data.sheet === 'signup') { const sheet = ss.getSheetByName('signups'); const lastRow = Math.max(sheet.getLastRow(), 1); sheet.getRange(lastRow+1, 1).setValue(data.name); sheet.getRange(lastRow+1, 2).setValue(data.email); sheet.getRange(lastRow+1, 3).setValue(data.phone); sheet.getRange(lastRow+1, 4).setValue(data.role); sheet.getRange(lastRow+1, 5).setValue(data.message); sheet.getRange(lastRow+1, 6).setValue(new Date().toLocaleString()); MailApp.sendEmail({ to: NOTIFY_EMAIL, subject: '🌱 New Signup — ' + data.role, htmlBody: `

` }); } // ── ANNOUNCEMENT ── if (data.sheet === 'announcement') { const sheet = ss.getSheetByName('announcement'); const rows = sheet.getDataRange().getValues(); rows.forEach((row, i) => { if (row[0] === 'text') sheet.getRange(i+1, 2).setValue(data.text || ''); if (row[0] === 'active') sheet.getRange(i+1, 2).setValue(data.active); if (row[0] === 'linkLabel') sheet.getRange(i+1, 2).setValue(data.linkLabel || ''); if (row[0] === 'linkTarget') sheet.getRange(i+1, 2).setValue(data.linkTarget || ''); }); } return ContentService.createTextOutput(JSON.stringify({ success: true })) .setMimeType(ContentService.MimeType.JSON); } catch(err) { Logger.log('doPost error: ' + err.toString()); return ContentService.createTextOutput(JSON.stringify({ success: false, error: err.toString() })) .setMimeType(ContentService.MimeType.JSON); } }
